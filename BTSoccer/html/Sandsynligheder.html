function (beta, theta, x, i, j)
{
    VTU = 0
    VTU = c((styrker[i])/(styrker[i] + theta * styrker[j]), ((styrker[j])/(styrker[i] *
        theta + styrker[j])), ((styrker[i] * styrker[j] * (theta^2 -
        1))/((styrker[i] + theta * styrker[j]) * (styrker[i] *
        theta + styrker[j]))))
    return(VTU)
  }
